// precedance and associativity of operators in java

[] ()                               left to right
! ~ ++ -- + - (cast) new            right to left
* / %                               left to right
+ -                                 left to right
<< >> >>>                           left to right
< <= > >= instanceof                left to right
== !=                               left to right
&                                   left to right
^                                   left to right
|                                   left to right
&&                                  left to right
||                                  left to right
?:                                  right to left
= += -= *= /= %= &= ^= |= <<= >>=   right to left
,                                   left to right
